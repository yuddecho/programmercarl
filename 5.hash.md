### 242 有效字母异位词

- https://leetcode.cn/problems/valid-anagram/
- 由于是单词, 所以只有26个字母, 直接用数组来统计就好了, 还可以 排序
- 而且在开始可以判断长度是否相等
- 先审题: 我是用map来写的

```c++
// 排序比大小
class Solution {
public:
    bool isAnagram(string s, string t) {
        if (s.length() != t.length()) {
            return false;
        }
        sort(s.begin(), s.end());
        sort(t.begin(), t.end());
        return s == t;
    }
};

// 数据组统计
class Solution {
public:
    bool isAnagram(string s, string t) {
        if (s.length() != t.length()) {
            return false;
        }
        vector<int> table(26, 0);
        for (auto& ch: s) {
            table[ch - 'a']++;
        }
        for (auto& ch: t) {
            table[ch - 'a']--;
            if (table[ch - 'a'] < 0) {
                return false;
            }
        }
        return true;
    }
};
```

```c++
#include<iostream>
#include<string>
#include<unordered_map>

using std::cout;
using std::endl;

using std::string;
using std::unordered_map;

class Solution
{
public:
    bool isAnagram(string s, string t)
    {
        unordered_map<char, int> umap;

        for (auto iter = s.begin(); iter != s.end(); ++iter)
        {
            umap[*iter] = umap[*iter] + 1;
        }
        
        for (auto iter = t.begin(); iter != t.end(); ++iter)
        {
            umap[*iter] = umap[*iter] - 1;
        }

        for (auto iter = umap.begin(); iter != umap.end(); ++iter)
        {
            if (iter->second != 0)
            {
                return false; 
            }
        }

        return true;
    }
};


int main(void)
{
    string ss = "ljljljl";
    string tt = "ljljljl";

    Solution s;
    cout << s.isAnagram(ss, tt) << endl;

    return 0;
}
```
